# cal.sh

#!/bin/bash

while [ $# -ge 3 ]
do
	if [ -n "$2" ] && [ -n "$3" ]
	then
		while getopts asmd opt
		do
			case "$opt" in
			a)
				echo $[$2+$3];;
			s)
				echo $[$2-$3];;
			m)
				echo $[$2*$3];;
			d)
				echo $(echo "scale=4;$2/$3"|bc);;
			*)
				echo "Unknown option:$opt";;
			esac
		done
	else
		echo "Useage: $0 [[-asmd] num1 num2]"
	fi
	shift 3
	$0 $1 $2 $3
done
